CREATE OR REPLACE TRIGGER trg_block_business_days
BEFORE INSERT OR UPDATE OR DELETE ON Sales
FOR EACH ROW
DECLARE
    v_count NUMBER;
BEGIN

    SELECT COUNT(*)
    INTO v_count
    FROM Holidays
    WHERE holiday_date = TRUNC(SYSDATE);

    IF v_count > 0 THEN
        RAISE_APPLICATION_ERROR(
            -20001,
            'Transactions are not allowed on public holidays'
        );
    END IF;

END;
/
